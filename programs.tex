\subsection{How to Compile}

\begin{frame}
\frametitle{How to Compile}
\begin{itemize}
    \item \texttt{latex}
    \begin{itemize}
      \item \texttt{latex index.tex}
    \end{itemize}
    \item \texttt{pdflatex}
    \begin{itemize}
        \item \texttt{pdflatex index.tex}
    \end{itemize}
    \item \texttt{latexmk}
    \begin{itemize}
        \item latexmk -pdf index.tex
        \item latexmk -c
    \end{itemize}
\end{itemize}
\end{frame}
